# DOS to Linux Command Mapping
# Format: <DOS command> = <Linux command>

# File operations
dir = ls -la
type = cat
copy = cp
xcopy = cp -r
move = mv
rename = mv
del = rm
erase = rm
rmdir = rmdir
rd = rmdir
md = mkdir

# System operations
cls = clear
ver = uname -a
date = date
time = date +%T
echo = echo
pause = read -p "Press Enter to continue..." dummy

# Network commands
ping = ping
ipconfig = ifconfig
netstat = netstat

# Process commands
tasklist = ps aux
taskkill = kill

# File system
chkdsk = fsck
diskpart = fdisk
format = mkfs

# Other
find = grep
sort = sort
more = more
help = man
